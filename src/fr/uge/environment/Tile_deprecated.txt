package fr.uge.environment;

import java.util.Set;

public sealed interface Tile permits KeystoneTile, HabitatTile, EmptyTile {
  /* abstract methods */
  Set<WildlifeType> animals();
  TileType leftHabitat();
  TileType rightHabitat();
  // WildlifeType[] getPossibleAnimals();
  // boolean canBePlaced(WildlifeType token);
}


  @Override
  public List<Cell> getNeighbors(Cell cell) {
    Objects.requireNonNull(cell);
    return Constants.SQUARE_DIRECTION_DIFFERENCES.stream()
        .map(diff -> getOneNeighbor(cell, diff))
        .filter(Objects::nonNull)
        .collect(Collectors.toUnmodifiableList());
  }
